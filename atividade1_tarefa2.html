Automação do Processo de Importação

A partir de um banco de dados criado para a atividade, é realizado o processo de criação automática da estrutura com as tabelas 'produtos', 'NCM' e 'vendas'. Tais tabelas deverão receber os dados constantes dos respectivos arquivos .csv. A cada execução do processo, as tabelas são excluídas e recriadas. 

-- colocando em uso o banco aula
USE aula;

-- apagando produtos se existir
DROP TABLE IF EXISTS produtos;

-- criando a tabela produtos
CREATE TABLE produtos (
    idproduto INT,
    descricao VARCHAR(255),
    unidade CHAR(255),
    ncm VARCHAR(255)
);

-- apagando NCM se existir
DROP TABLE IF EXISTS NCM;

-- criando a tabela NCM
CREATE TABLE NCM (
    NCM CHAR(MAX),
    lixo1 VARCHAR(MAX),
    descricao VARCHAR(MAX),
    lixo2 VARCHAR(MAX)
);

-- excluindo vendas se existir
DROP TABLE IF EXISTS vendas;

-- criando vendas
CREATE TABLE vendas (
    data DATE,
    produto VARCHAR(255),
    quantidade VARCHAR(255),
    total VARCHAR(255)
);
